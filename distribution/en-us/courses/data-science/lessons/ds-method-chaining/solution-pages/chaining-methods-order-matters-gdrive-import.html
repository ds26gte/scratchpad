fun kilograms(r): r["pounds"] / 2.2     end
fun is-heavy(r):  r["kilos"] > 25       end

t.order-by("kilos", true)

1 (D)

A

t.filter(is-female)
  .build-column("kilos", kilograms)

t.build-column("kilos", kilograms)
  .filter(is-heavy)
